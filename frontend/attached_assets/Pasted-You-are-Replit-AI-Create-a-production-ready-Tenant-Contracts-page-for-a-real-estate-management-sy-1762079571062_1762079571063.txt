## Data Model (TypeScript)
type Money = { currency: "SAR"; amount: number };
type ContractStatus = "ACTIVE" | "EXPIRING" | "TERMINATED" | "PAST";
type PaymentStatus = "PAID" | "OVERDUE" | "UPCOMING";

interface PaymentItem {
  id: string;
  installmentNo: number;
  periodLabel: string; // "Nov 2025"
  dueDate: string;     // ISO
  amount: Money;
  status: PaymentStatus;
  paidOn?: string;     // ISO
  receiptUrl?: string; // mock link
}

interface ContractDocument {
  id: string;
  type: "CONTRACT" | "ADDENDUM" | "RECEIPT";
  title: string;
  url: string;     // mock
  uploadedAt: string;
}

interface Contract {
  id: string;
  propertyName: string;
  unitLabel: string;
  leaseStart: string; // ISO
  leaseEnd: string;   // ISO
  monthlyRent: Money;
  paymentCycle: "MONTHLY";
  securityDeposit: Money;
  status: ContractStatus;
  nextPayment?: PaymentItem;
  payments: PaymentItem[];
  documents: ContractDocument[];
}
